—Exercises (E#) generally progress from less complex to more complex


—E1
SELECT cities.name AS city, countries.name AS country, region
FROM cities
INNER JOIN countries
ON cities.country_code = countries.code;

—E2
SELECT c.code AS country_code, name, year, inflation_rate
FROM countries AS c
INNER JOIN economies AS e
ON c.code = e.code;

—E3
SELECT c.code, name, region, e.year, fertility_rate, unemployment_rate
FROM countries AS c
INNER JOIN populations AS p
ON c.code = p.country_code
INNER JOIN economies AS e
ON c.code = e.code AND p.year = e.year;

—E4
SELECT c.name AS country, continent, l.name AS language, official
FROM countries AS c
INNER JOIN languages AS l
USING (code);

—E5
SELECT p1.country_code, p1.size AS size2010, p2.size AS size2015, (p2.size - p1.size) / p1.size * 100.0  AS growth_perc
FROM populations AS p1
INNER JOIN populations AS p2
ON p1.country_code = p2.country_code AND p1.year = p2.year - 5;

—E6
SELECT name, continent, code, surface_area,
    CASE WHEN surface_area > 2000000
            THEN 'large'
       WHEN surface_area > 350000
            THEN 'medium'
       ELSE 'small' END 
       AS geosize_group
FROM countries;

—E7
SELECT country_code, size,
    CASE WHEN size > 50000000
        THEN 'large'
    WHEN size > 1000000
        THEN 'medium'
    ELSE 'small' END 
    AS popsize_group
INTO pop_plus
FROM populations
WHERE year = 2015;

SELECT name, continent, geosize_group, popsize_group
FROM countries_plus AS c
INNER JOIN pop_plus AS p
ON p.country_code = c.code
ORDER BY geosize_group DESC;

—E8
SELECT c.name AS country, local_name, l.name AS language, percent
FROM countries AS c
LEFT JOIN languages AS l
ON c.code = l.code
ORDER BY country DESC;

—E9
SELECT region, AVG(gdp_percapita) AS avg_gdp
FROM countries AS c
LEFT JOIN economies AS e
ON c.code = e.code
WHERE year = 2010
GROUP BY region
ORDER BY avg_gdp DESC;

—E10
SELECT cities.name AS city, urbanarea_pop, countries.name AS country,
       indep_year, languages.name AS language, percent
FROM languages 
RIGHT JOIN countries 
ON languages.code = countries.code
RIGHT JOIN cities 
ON countries.code = cities.country_code
ORDER BY city, language;

—E11
SELECT name AS country, code, region, basic_unit
FROM countries
FULL JOIN currencies
USING (code)
WHERE region = 'North America' OR region IS NULL
ORDER BY region;

—E12
SELECT countries.name, code, languages.name AS language
FROM languages
FULL JOIN countries
USING (code)
WHERE countries.name LIKE 'V%' OR countries.name IS NULL
ORDER BY countries.name;

—E13
SELECT c1.name AS country, region, l.name AS language,
       basic_unit, frac_unit
FROM countries AS c1
FULL JOIN languages AS l
USING (code)
FULL JOIN currencies AS c2
USING (code)
WHERE region LIKE 'M%esia';

—E14
SELECT c.name AS city, l.name AS language
FROM cities AS c
CROSS JOIN languages AS l
WHERE c.name LIKE 'Hyder%’;

—E15
SELECT name AS country, region, life_expectancy AS life_exp
FROM countries 
LEFT JOIN populations
ON countries.code = populations.country_code
WHERE year = 2010
ORDER BY life_exp
LIMIT 5;

—E16
SELECT *
FROM economies2010
UNION 
SELECT *
FROM economies2015
ORDER BY code, year;

—E17
SELECT country_code
FROM cities
UNION
SELECT code
FROM currencies
ORDER BY country_code;

—E18
SELECT code, year
FROM economies
UNION ALL
SELECT country_code, year
FROM populations
ORDER BY code, year;

—E19
SELECT code, year
FROM economies
INTERSECT
SELECT country_code, year
FROM populations
ORDER BY code, year;

—E20
SELECT name
FROM countries
INTERSECT
SELECT name
FROM cities;

—E21
SELECT name
FROM cities
EXCEPT
SELECT capital
FROM countries
ORDER BY name;

—E22
SELECT capital
FROM countries
EXCEPT
SELECT name
FROM cities
ORDER BY capital;

—E23
SELECT DISTINCT name
FROM languages
WHERE code IN
    (SELECT code
    FROM countries
    WHERE region = 'Middle East')
ORDER BY name;

—E24
SELECT code, name 
FROM countries 
WHERE continent = 'Oceania' AND code NOT IN
    (SELECT code
    FROM currencies);

—E25
SELECT c1.name
FROM cities AS c1
WHERE country_code IN
    (SELECT e.code
    FROM economies AS e
    UNION
    SELECT c2.code
    FROM currencies AS c2
    EXCEPT
    SELECT p.country_code
    FROM populations AS p);
